--- --- Custom Data

Using Custom Data

CSV file

{%- for anime in site.data.data-csv -%}

{{anime.rank}}: {{anime.title}}

{%- endfor -%}

JSON file

{%- for anime in site.data.data-json.items -%}

{{anime.rank}}: {{anime.title}}

{%- endfor -%}

YML file

{%- for anime in site.data.data-yml.items -%}

{{anime.rank}}: {{anime.title}}

{%- if anime.subitems -%} {%- for subitem in anime.subitems -%}

    Sub Rank: {{subitem.rank}}. {{subitem.title}}

{%- endfor -%} {%- endif -%} {%- endfor -%}

YML file Sorted by Rank

{%- assign animes = site.data.data-yml.items | sort: "rank" -%} {%- for anime in animes -%}

{{anime.rank}}: {{anime.title}}

{%- if anime.subitems -%} {%- for subitem in anime.subitems -%}

    Sub Rank: {{subitem.rank}}. {{subitem.title}}

{%- endfor -%} {%- endif -%} {%- endfor -%}
